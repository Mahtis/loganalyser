19      public ExperimentInfo loadFromFile(String filename) {
20          List<String> lines = FileReader.readFile(filename);
21          if (!lines.isEmpty()) {
22              ExperimentInfo info = new ExperimentInfo();
23              //button codes and definitions located on lines 0 and 1
24              List<String> respCodes = Arrays.asList(lines.get(0).split("\t"));
25              info.setResponseCodes(respCodes);
26              List<String> respNames = Arrays.asList(lines.get(1).split("\t"));
27              info.setResponseNames(respNames);
28              // conditions and response mapping on lines 2 ->
29              List<String> conds = new ArrayList<>();
30              for (int i = 2; i < lines.size(); i++) {
31                  List<String> line = Arrays.asList(lines.get(i).split("\t"));
32                  // new response mapping with condition from column 0
33                  ResponseMapping mapping = new ResponseMapping(line.get(0));
34                  // set this also to ExperimentInfo for easier access.
35                  conds.add(line.get(0));
36                  // number of responses in column 1
37                  mapping.setnOfResponses(Integer.parseInt(line.get(1)));
38                  // correct response for each response in columns 2->
39                  List<List<String>> corrResps = new ArrayList<>();
40                  for (int j = 2; j < line.size(); j++) {
41                      // multiple correct responses for each response-condition are separated by ,
42                      // e.g. 1,2 2   1,2,3 correct for the first is 1 or 2
43                      // correct for second is 2, correct for third is 1, 2 or 3.
44                      // One resp per condition would be "cond    1   1" and the
45                      // list would be mapping.get(0).get(0) == 1, whereas the above
46                      // could be mapping.get(2).get(2) == 3.
47                      corrResps.add(Arrays.asList(line.get(j).split(",")));
48                  }
49                  mapping.setCorrectResponses(corrResps);
50                  info.addResponseMapping(mapping);
51              }
52              info.setConditions(conds);
53              return info;
54          }
55          return null;
56      }

58      public void saveToFile(ExperimentInfo info, String location, String expName) throws IOException {
59          List<String> lines = new ArrayList<>();
60          List<ResponseMapping> mappings = info.getResponseMappings();
61          List<String> respCodes = info.getResponseCodes();
62          List<String> respNames = info.getResponseNames();
63          StringBuilder codeLine = new StringBuilder();
64          for (String code : respCodes) {
65              codeLine.append(code);
66              codeLine.append("\t");
67          }
68          lines.add(codeLine.toString().trim());
69          StringBuilder nameLine = new StringBuilder();
70          for (String name : respNames) {
71              nameLine.append(name);
72              nameLine.append("\t");
73          }
74          lines.add(nameLine.toString().trim());
75          for (ResponseMapping mapping : mappings) {
76              StringBuilder line = new StringBuilder();
77              line.append(mapping.getCondition());
78              line.append("\t");
79              line.append(mapping.getnOfResponses());
80              line.append("\t");
81              List<List<String>> corrs = mapping.getCorrectResponses();
82              for (int i = 0; i < corrs.size(); i++) {
83                  for (int j = 0; j < corrs.get(i).size(); j++) {
84                      line.append(corrs.get(i).get(j));
85                      line.append(",");
86                  }
87  
88              }
89              line.deleteCharAt(line.length() - 1);
90              lines.add(line.toString());
91          }
92          Path file = Paths.get(location + expName + ".txt");
93          Files.write(file, lines, Charset.forName("UTF-8"));
94  //Files.write(file, lines, Charset.forName("UTF-8"), StandardOpenOption.APPEND);
95      }
